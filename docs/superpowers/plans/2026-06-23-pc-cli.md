# `pc` CLI + MCP Server Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `cmd/pc` — a harness-agnostic CLI plus an MCP server — over a shared `internal/pcops` core, so any CLI coding agent participates in gate coordination by running a shell command or calling one MCP tool.

**Architecture:** `internal/pcops` wraps `pkg/gate` + `pkg/bus/sqlite` with one function per operation (`LoadConfig`, `Submit`, `Up`, `RunGate`, `Watch`, `GateStatus`). `cmd/pc` is two thin skins over it: subcommand/flag parsing (CLI) and an `mcp.Server` (`pc mcp`) whose tools call the same functions.

**Tech Stack:** Go 1.23, `github.com/modelcontextprotocol/go-sdk` v1.2.0, `gopkg.in/yaml.v3` v3.0.1, `modernc.org/sqlite` (already present), standard library.

// Package pcops implements the operations behind the pc CLI and MCP server:
// loading config and driving the gate over the SQLite bus. The CLI and MCP
// server are thin skins over these functions.
package pcops

import (
	"fmt"
	"os"
	"time"

	"gopkg.in/yaml.v3"
)

// GateDef is one gate's definition in .pc.yaml.
type GateDef struct {
	Required []string `yaml:"required"`
	Runner   string   `yaml:"runner"`
	Run      string   `yaml:"run"`
}

// Config is the resolved .pc.yaml.
type Config struct {
	DB            string
	Gates         map[string]GateDef
	SubmitTimeout time.Duration
}

type rawConfig struct {
	DB       string             `yaml:"db"`
	Gates    map[string]GateDef `yaml:"gates"`
	Defaults struct {
		SubmitTimeout string `yaml:"submit_timeout"`
	} `yaml:"defaults"`
}

// LoadConfig reads and resolves .pc.yaml. $PC_DB overrides the db path;
// SubmitTimeout defaults to 5m.
func LoadConfig(path string) (Config, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return Config{}, fmt.Errorf("read config %q: %w", path, err)
	}
	var raw rawConfig
	if err := yaml.Unmarshal(data, &raw); err != nil {
		return Config{}, fmt.Errorf("parse config %q: %w", path, err)
	}
	cfg := Config{DB: raw.DB, Gates: raw.Gates, SubmitTimeout: 5 * time.Minute}
	if cfg.Gates == nil {
		cfg.Gates = map[string]GateDef{}
	}
	if raw.Defaults.SubmitTimeout != "" {
		d, err := time.ParseDuration(raw.Defaults.SubmitTimeout)
		if err != nil {
			return Config{}, fmt.Errorf("parse submit_timeout: %w", err)
		}
		cfg.SubmitTimeout = d
	}
	if env := os.Getenv("PC_DB"); env != "" {
		cfg.DB = env
	}
	if cfg.DB == "" {
		return Config{}, fmt.Errorf("no db configured (set `db:` in config or $PC_DB)")
	}
	return cfg, nil
}

// Gate looks up a gate definition by id.
func (c Config) Gate(id string) (GateDef, error) {
	g, ok := c.Gates[id]
	if !ok {
		return GateDef{}, fmt.Errorf("unknown gate %q (not in config)", id)
	}
	return g, nil
}
```

// Submit declares agentName ready for gateID at version, then blocks until the
// gate's verdict (or cfg.SubmitTimeout). The subscription uses the agent's
// durable identity, so repeated submits resume past prior verdicts.
func Submit(ctx context.Context, cfg Config, gateID, agentName, version string) (gate.Verdict, error) {
	if _, err := cfg.Gate(gateID); err != nil {
		return gate.Verdict{}, err
	}
	b, err := sqlite.Open(ctx, cfg.DB)
	if err != nil {
		return gate.Verdict{}, err
	}
	defer b.Close()

	a, err := agent.New(ctx, b, agentName, []string{gate.Topic(gateID)})
	if err != nil {
		return gate.Verdict{}, err
	}

	waitCtx, cancel := context.WithTimeout(ctx, cfg.SubmitTimeout)
	defer cancel()

	verdicts := make(chan gate.Verdict, 1)
	a.On(protocol.IntentInform, func(_ context.Context, _ *agent.Agent, m protocol.Message) *protocol.Message {
		if g, _ := m.Body["gate"].(string); g == gateID {
			if _, ok := m.Body["passed"]; ok {
				passed, _ := m.Body["passed"].(bool)
				text, _ := m.Body["text"].(string)
				select {
				case verdicts <- gate.Verdict{GateID: gateID, Passed: passed, Detail: text}:
				default:
				}
			}
		}
		return nil
	})
	go a.Run(waitCtx)

	if err := gate.Ready(waitCtx, a, gateID, version); err != nil {
		return gate.Verdict{}, err
	}

	select {
	case v := <-verdicts:
		return v, nil
	case <-waitCtx.Done():
		if ctx.Err() != nil {
			return gate.Verdict{}, ctx.Err()
		}
		return gate.Verdict{}, fmt.Errorf("timed out after %s waiting for gate %q verdict", cfg.SubmitTimeout, gateID)
	}
}
```

// Up hosts the coordinator for every configured gate until ctx is cancelled.
func Up(ctx context.Context, cfg Config) error {
	b, err := sqlite.Open(ctx, cfg.DB)
	if err != nil {
		return err
	}
	defer b.Close()

	topics := make([]string, 0, len(cfg.Gates))
	for id := range cfg.Gates {
		topics = append(topics, gate.Topic(id))
	}
	gk, err := agent.New(ctx, b, "pc-coordinator", topics)
	if err != nil {
		return err
	}
	coord := gate.NewCoordinator(gk)
	for id, def := range cfg.Gates {
		coord.Register(gate.Spec{ID: id, Required: def.Required, Runner: def.Runner})
	}
	go gk.Run(ctx)
	<-ctx.Done()
	return nil
}
```

// RunGate hosts the runner for gateID until ctx is cancelled. When the gate
// opens it runs the gate's Run command (or override, if non-empty); exit 0 maps
// to a passing verdict, non-zero to a failing one with the command output as detail.
func RunGate(ctx context.Context, cfg Config, gateID string, override []string) error {
	def, err := cfg.Gate(gateID)
	if err != nil {
		return err
	}
	cmdline := override
	if len(cmdline) == 0 {
		if def.Run == "" {
			return fmt.Errorf("gate %q has no `run` command and no override given", gateID)
		}
		cmdline = []string{"sh", "-c", def.Run}
	}
	b, err := sqlite.Open(ctx, cfg.DB)
	if err != nil {
		return err
	}
	defer b.Close()

	r, err := agent.New(ctx, b, def.Runner, nil)
	if err != nil {
		return err
	}
	gate.ServeRunner(r, func(ctx context.Context, id string, versions map[string]string) gate.Verdict {
		out, runErr := runCommand(ctx, cmdline)
		if runErr != nil {
			return gate.Verdict{GateID: id, Passed: false, Detail: lastLine(out, runErr)}
		}
		return gate.Verdict{GateID: id, Passed: true, Versions: versions}
	})
	go r.Run(ctx)
	<-ctx.Done()
	return nil
}

func runCommand(ctx context.Context, cmdline []string) (string, error) {
	cmd := exec.CommandContext(ctx, cmdline[0], cmdline[1:]...)
	var buf strings.Builder
	cmd.Stdout = &buf
	cmd.Stderr = &buf
	err := cmd.Run()
	return buf.String(), err
}

func lastLine(out string, err error) string {
	lines := strings.Split(strings.TrimSpace(out), "\n")
	for i := len(lines) - 1; i >= 0; i-- {
		if s := strings.TrimSpace(lines[i]); s != "" {
			return s
		}
	}
	return err.Error()
}
```

// Watch streams the activity on all configured gate topics to w — history first
// (replay) then live — until ctx is cancelled. It uses a per-process observer
// identity so each invocation replays from the start of the log.
func Watch(ctx context.Context, cfg Config, w io.Writer) error {
	b, err := sqlite.Open(ctx, cfg.DB, sqlite.WithReplayFromZero())
	if err != nil {
		return err
	}
	defer b.Close()

	topics := make([]string, 0, len(cfg.Gates))
	for id := range cfg.Gates {
		topics = append(topics, gate.Topic(id))
	}
	name := "pc-watch-" + strconv.Itoa(os.Getpid())
	ch, err := b.Subscribe(ctx, name, topics)
	if err != nil {
		return err
	}
	for {
		select {
		case <-ctx.Done():
			return nil
		case m, ok := <-ch:
			if !ok {
				return nil
			}
			fmt.Fprintln(w, formatTurn(m))
		}
	}
}

func formatTurn(m protocol.Message) string {
	to := m.To.Agent
	if to == "" {
		to = "#" + m.To.Topic
	}
	text, _ := m.Body["text"].(string)
	return fmt.Sprintf("%-12s --%-8s--> %-14s %s", m.From.Agent, m.Intent, to, text)
}

func run(args []string, stdout, stderr io.Writer) int {
	if len(args) == 0 {
		fmt.Fprintln(stderr, usage)
		return 2
	}
	ctx, stop := signal.NotifyContext(context.Background(), os.Interrupt)
	defer stop()

	switch args[0] {
	case "submit":
		return cmdSubmit(ctx, args[1:], stdout, stderr)
	case "up":
		return cmdUp(ctx, args[1:], stderr)
	case "run-gate":
		return cmdRunGate(ctx, args[1:], stderr)
	case "watch":
		return cmdWatch(ctx, args[1:], stdout, stderr)
	default:
		fmt.Fprintf(stderr, "unknown command %q\n%s\n", args[0], usage)
		return 2
	}
}

		fmt.Fprintln(stderr, err)

	}
	return cfg, true
}

	if passed {
		return 0
	}

func cmdSubmit(ctx context.Context, args []string, stdout, stderr io.Writer) int {
	fs := flag.NewFlagSet("submit", flag.ContinueOnError)
	fs.SetOutput(stderr)

	asName := fs.String("as", "", "agent identity (or $PC_AGENT)")

		return 2

		fmt.Fprintln(stderr, "submit: --gate is required")
		return 2

	if !ok {
		return 2
	}
	name, err := resolveAgent(*asName)
	if err != nil {
		fmt.Fprintln(stderr, err)
		return 2

// Status is the last broadcast verdict for a gate.
type Status struct {
	Gate   string `json:"gate"`
	Found  bool   `json:"found"`
	Passed bool   `json:"passed"`
	Detail string `json:"detail"`
}

// GateStatus reports the last broadcast verdict for gateID, best-effort: it
// replays the gate topic and keeps the most recent verdict inform seen within a
// short drain window.
func GateStatus(ctx context.Context, cfg Config, gateID string) (Status, error) {
	if _, err := cfg.Gate(gateID); err != nil {
		return Status{}, err
	}
	b, err := sqlite.Open(ctx, cfg.DB, sqlite.WithReplayFromZero())
	if err != nil {
		return Status{}, err
	}
	defer b.Close()

	name := "pc-status-" + strconv.Itoa(os.Getpid())
	ch, err := b.Subscribe(ctx, name, []string{gate.Topic(gateID)})
	if err != nil {
		return Status{}, err
	}
	st := Status{Gate: gateID}
	drain, cancel := context.WithTimeout(ctx, 500*time.Millisecond)
	defer cancel()
	for {
		select {
		case <-drain.Done():
			return st, nil
		case m, ok := <-ch:
			if !ok {
				return st, nil
			}
			if g, _ := m.Body["gate"].(string); g == gateID {
				if passed, ok := m.Body["passed"].(bool); ok {
					text, _ := m.Body["text"].(string)
					st.Found, st.Passed, st.Detail = true, passed, text
				}
			}
		}
	}
}

package main

import (
	"context"
	"flag"
	"fmt"
	"io"

	"github.com/modelcontextprotocol/go-sdk/mcp"

	"github.com/KJFromMicromonic/parallel-consciousness/internal/pcops"
)

type submitInput struct {
	Gate    string `json:"gate" jsonschema:"the gate id to submit to"`
	Version string `json:"version,omitempty" jsonschema:"version token (defaults to git HEAD)"`
	Agent   string `json:"agent,omitempty" jsonschema:"agent identity (defaults to $PC_AGENT)"`
}
type submitOutput struct {
	Passed   bool              `json:"passed"`
	Detail   string            `json:"detail"`
	Versions map[string]string `json:"versions"`
}

type statusInput struct {
	Gate string `json:"gate" jsonschema:"the gate id to report"`
}

// submitTool is the testable core of the submit_to_gate MCP tool.
func submitTool(ctx context.Context, cfg pcops.Config, in submitInput) (submitOutput, error) {
	agentName, err := resolveAgent(in.Agent)
	if err != nil {
		return submitOutput{}, err
	}
	version, err := resolveVersion(ctx, in.Version)
	if err != nil {
		return submitOutput{}, err
	}
	v, err := pcops.Submit(ctx, cfg, in.Gate, agentName, version)
	if err != nil {
		return submitOutput{}, err
	}
	return submitOutput{Passed: v.Passed, Detail: v.Detail, Versions: v.Versions}, nil
}

func cmdMCP(ctx context.Context, args []string, stderr io.Writer) int {
	fs := flag.NewFlagSet("mcp", flag.ContinueOnError)
	fs.SetOutput(stderr)
	cfgPath := fs.String("config", ".pc.yaml", "config file")
	if err := fs.Parse(args); err != nil {
		return 2
	}
	cfg, ok := loadCfg(*cfgPath, stderr)
	if !ok {
		return 2
	}

	server := mcp.NewServer(&mcp.Implementation{Name: "parallel-consciousness", Version: "v0.1.0"}, nil)

	mcp.AddTool(server, &mcp.Tool{
		Name:        "submit_to_gate",
		Description: "Declare ready for a gate and block until the cross-service test verdict.",
	}, func(ctx context.Context, _ *mcp.CallToolRequest, in submitInput) (*mcp.CallToolResult, submitOutput, error) {
		out, err := submitTool(ctx, cfg, in)
		return nil, out, err
	})

	mcp.AddTool(server, &mcp.Tool{
		Name:        "gate_status",
		Description: "Report the last known verdict for a gate.",
	}, func(ctx context.Context, _ *mcp.CallToolRequest, in statusInput) (*mcp.CallToolResult, pcops.Status, error) {
		st, err := pcops.GateStatus(ctx, cfg, in.Gate)
		return nil, st, err
	})

	if err := server.Run(ctx, &mcp.StdioTransport{}); err != nil {
		fmt.Fprintln(stderr, "mcp:", err)
		return 1
	}
	return 0
}
